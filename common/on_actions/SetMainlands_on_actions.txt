# Set Mainland States as Array Varaibles
# Mainly used for unifications

on_actions = {
	on_startup = {
		effect = {
			every_state = {
				# Turkey Mainland (Cyprus not included)
				if = {
					limit = {
						OR = {
							is_core_of = ANK # Not owned, because starts in war
							is_owned_by = TSK
							is_owned_by = TFT
						}
					}
					
					add_to_array = {
						array = TUR.mainland_states
						value = THIS #.id
					}
				}
			}
		}
	}
}
